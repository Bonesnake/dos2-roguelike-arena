// Cleans regions, locks waypoints and removes regions from the active regions list.
PROC
PROC_RoguelikeArena_ResetRegions()
AND
DB_RoguelikeArena_ActiveRegions(_Region, _Level, _Index)
THEN
NOT DB_RoguelikeArena_ActiveRegions(_Region, _Level, _Index);

// Resets region pool and musics, and saves teleporter pyramids.
PROC
PROC_RoguelikeArena_ResetRegions()
THEN
PROC_RoguelikeArena_IntegerPoolClear("RegionPool");
PROC_RoguelikeArena_IntegerPoolAddRegions();
PROC_RoguelikeArena_SaveTeleporterPyramids();

// Selects a new region and prepares it for population.
PROC
PROC_RoguelikeArena_StartNewRegion((INTEGER)_Level)
AND
QRY_RoguelikeArena_IntegerPoolSelectAndRemoveRandom("RegionPool")
AND
DB_RoguelikeArena_IntegerPoolSelectedValue("RegionPool", _Region)
AND
SysCount("DB_RoguelikeArena_ActiveRegions", 3, _CurrentIndex)
AND
IntegerSum(_CurrentIndex, 1, _NextIndex)
THEN
PROC_RoguelikeArena_GeneratePath(_Region, 4);
DB_RoguelikeArena_ActiveRegions(_Region, _Level, _NextIndex);
PROC_RoguelikeArena_GenerateEncounters(_Region);

// Adds available regions to the region pool.
PROC
PROC_RoguelikeArena_IntegerPoolAddRegions()
AND
DB_RoguelikeArena_Regions(_Region, _)
THEN
PROC_RoguelikeArena_IntegerPoolAdd("RegionPool", _Region);
