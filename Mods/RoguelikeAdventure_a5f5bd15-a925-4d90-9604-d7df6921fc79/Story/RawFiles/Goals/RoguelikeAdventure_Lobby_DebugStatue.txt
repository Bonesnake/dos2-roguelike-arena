// Levels up the character.
IF
ObjectFlagSet("RoguelikeAdventure_DebugStatue_WasUsed", _Character, _)
AND
ObjectGetFlag(_Character, "RoguelikeAdventure_DebugStatue_LevelUp", 1)
THEN
CharacterLevelUp((CHARACTERGUID)_Character);

// Adds gold to the character.
IF
ObjectFlagSet("RoguelikeAdventure_DebugStatue_WasUsed", _Character, _)
AND
ObjectGetFlag(_Character, "RoguelikeAdventure_DebugStatue_AddGold", 1)
THEN
CharacterAddGold((CHARACTERGUID)_Character, 99999);

// Applies buffs to the character.
IF
ObjectFlagSet("RoguelikeAdventure_DebugStatue_WasUsed", _Character, _)
AND
ObjectGetFlag(_Character, "RoguelikeAdventure_DebugStatue_ApplyBuffs", 1)
AND
DB_RoguelikeAdventure_DebugStatueBuffs(_Status)
THEN
RemoveStatus(_Character, _Status);
ApplyStatus(_Character, _Status, -1.0, 1, NULL_00000000-0000-0000-0000-000000000000);

// Changes the preset of the character.
IF
ObjectFlagSet("RoguelikeAdventure_DebugStatue_WasUsed", _Character, _)
AND
DB_RoguelikeAdventure_DebugStatuePresets(_Preset)
AND
StringConcatenate("RoguelikeAdventure_DebugStatue_", _Preset, _PresetFlag)
AND
ObjectGetFlag(_Character, _PresetFlag, 1)
THEN
CharacterApplyHenchmanPreset((CHARACTERGUID)_Character, _Preset);
