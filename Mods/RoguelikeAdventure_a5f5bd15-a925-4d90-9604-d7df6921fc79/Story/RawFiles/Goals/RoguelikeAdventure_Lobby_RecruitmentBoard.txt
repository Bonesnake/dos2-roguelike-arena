PROC
PROC_RoguelikeAdventure_InitCompanionCharacters()
AND
DB_RoguelikeAdventure_AvailableCompanions(_Companion)
THEN
SetOnStage(_Companion, 0);

IF
DB_RoguelikeAdventure_CompanionsHelper(_Companion, _Recruiter)
AND
NOT DB_RoguelikeAdventure_Companions(_Companion, _Recruiter)
THEN
PROC_RoguelikeAdventure_GuidStringPoolAdd("AvailableCompanionPool", _Companion);

// Recruits a companion character with the help of flags set by the recruitment board.
IF
ObjectFlagSet("RoguelikeAdventure_RecruitmentBoard_WasUsed", _Player, _)
AND
QRY_RoguelikeAdventure_GuidStringPoolSelectAndRemoveRandom("AvailableCompanionPool")
AND
DB_RoguelikeAdventure_GuidStringPoolSelectedValue("AvailableCompanionPool", _Companion)
AND
DB_RoguelikeAdventure_CompanionTemplates(_Identity, _Template)
AND
StringConcatenate("RoguelikeAdventure_RecruitmentBoard_", _Identity, _IdentityFlag)
AND
ObjectGetFlag(_Player, _IdentityFlag, 1)
AND
DB_RoguelikeAdventure_CompanionPresets(_Preset)
AND
StringConcatenate("RoguelikeAdventure_RecruitmentBoard_", _Preset, _PresetFlag)
AND
ObjectGetFlag(_Player, _PresetFlag, 1)
THEN
SetOnStage(_Companion, 1);
PROC_RoguelikeAdventure_MakeCompanion((CHARACTERGUID)_Companion, _Template, _Preset, (CHARACTERGUID)_Player);

// Sets a flag when no companions can be recruited.
IF
DB_RoguelikeAdventure_Companions(_, _)
AND
SysCount("DB_RoguelikeAdventure_Companions", 2, _CompanionsCount)
AND
SysCount("DB_RoguelikeAdventure_AvailableCompanions", 1, _AvailableCompanionsCount)
AND
_CompanionsCount >= _AvailableCompanionsCount
THEN
GlobalSetFlag("RoguelikeAdventure_RecruitmentBoard_NoCompanionAvailable");

// Clears a flag when at least one companion can be recruited.
IF
StoryEvent(_, "RoguelikeAdventure_Companion_Dismissed")
AND
GlobalGetFlag("RoguelikeAdventure_RecruitmentBoard_NoCompanionAvailable", 1)
THEN
GlobalClearFlag("RoguelikeAdventure_RecruitmentBoard_NoCompanionAvailable");

PROC
PROC_RoguelikeAdventure_GuidStringPoolAddAvailableCompanions()
AND
DB_RoguelikeAdventure_AvailableCompanions(_Companion)
THEN
PROC_RoguelikeAdventure_GuidStringPoolAdd("AvailableCompanionPool", _Companion);
