// Resets skill count.
QRY
QRY_RoguelikeAdventure_SetCurrentCharacterSkillCount((INTEGER)_Level, (INTEGER)_Rank)
THEN
DB_Singleton("RoguelikeAdventure_CurrentCharacterSkillCount", 0);

// Sets skill count. The formula is: Random(Min, Max) + (Level - 1) / (Interval - Rank) + Rank
QRY
QRY_RoguelikeAdventure_SetCurrentCharacterSkillCount((INTEGER)_Level, (INTEGER)_Rank)
AND
DB_RoguelikeAdventure_SkillCountParameters(_SkillCountMin, _SkillCountMax, _Interval)
AND
QRY_RoguelikeAdventure_RandomInclusive(_SkillCountMin, _SkillCountMax)
AND
DB_Singleton("RoguelikeAdventure_RandomResult", _RandomResult)
AND
IntegerSubtract(_Level, 1, _LevelCorrected)
AND
IntegerSubtract(_Interval, _Rank, _IntervalCorrected)
AND
IntegerDivide(_LevelCorrected, _IntervalCorrected, _Modifier)
AND
IntegerSum(_Modifier, _Rank, _ModifierCorrected)
AND
IntegerSum(_RandomResult, _ModifierCorrected, _SkillCount)
THEN
DB_Singleton("RoguelikeAdventure_CurrentCharacterSkillCount", _SkillCount);

// Resets source skill count.
QRY
QRY_RoguelikeAdventure_SetCurrentCharacterSourceSkillCount((INTEGER)_Level, (INTEGER)_Rank)
THEN
DB_Singleton("RoguelikeAdventure_CurrentCharacterSourceSkillCount", 0);

// Sets source skill count. Formula is: Enemy Level * (Chance Increase Per Level + Enemy Rank) - Chance Decrease Per Index * Skill Index
QRY
QRY_RoguelikeAdventure_SetCurrentCharacterSourceSkillCount((INTEGER)_Level, (INTEGER)_Rank)
AND
DB_RoguelikeAdventure_SourceSkillChanceParameters(_ChanceIncreasePerLevel, _ChanceDecreasePerIndex)
AND
IntegerSum(_ChanceIncreasePerLevel, _Rank, _ChanceIncreasePerLevelRankCorrected)
AND
IntegerProduct(_Level, _ChanceIncreasePerLevelRankCorrected, _ChanceIncrease)
AND
IntegerSum(_Rank, 1, _SourceSkillMaxCount)
AND
QRY_RoguelikeAdventure_CreateIndices(_SourceSkillMaxCount)
AND
DB_RoguelikeAdventure_CreatedIndices(_SkillIndex)
AND
IntegerProduct(_ChanceDecreasePerIndex, _SkillIndex, _ChanceDecrease)
AND
IntegerSubtract(_ChanceIncrease, _ChanceDecrease, _Chance)
AND
IntegerMin(_Chance, 90, _ChanceCapped)
AND
Random(100, _Random)
AND
_Random < _ChanceCapped
THEN
PROC_RoguelikeAdventure_IncreaseCurrentCharacterSourceSkillCount();

PROC
PROC_RoguelikeAdventure_IncreaseCurrentCharacterSourceSkillCount()
AND
DB_Singleton("RoguelikeAdventure_CurrentCharacterSourceSkillCount", _SkillCountCurrent)
AND
IntegerSum(_SkillCountCurrent, 1, _SkillCountNew)
THEN
DB_Singleton("RoguelikeAdventure_CurrentCharacterSourceSkillCount", _SkillCountNew);
//END_REGION

//REGION Source
//Sets source points for character. Formula is: Random(Min, Level / (36 / (Rank * Rank + Modifier)))
PROC
PROC_RoguelikeAdventure_CharacterSetSourcePoints((CHARACTERGUID)_Character, (INTEGER)_Level, (INTEGER)_Rank)
AND
IntegerProduct(_Rank, _Rank, _RankPow)
AND
IntegerSum(_RankPow, 2, _Max)
AND
IntegerDivide(36, _Max, _MaxDistribution)
AND
IntegerDivide(_Level, _MaxDistribution, _MaxDistributed)
AND
QRY_RoguelikeAdventure_RandomInclusive(0, _MaxDistributed)
AND
DB_Singleton("RoguelikeAdventure_RandomResult", _Points)
THEN
CharacterOverrideMaxSourcePoints(_Character, _Max);
CharacterAddSourcePoints(_Character, _Points);
//END_REGION
